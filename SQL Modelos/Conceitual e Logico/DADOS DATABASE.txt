CREATE TABLE PROFESSOR
(
    PROFNUMERO INTEGER PRIMARY KEY NOT NULL AUTO_INCREMENT,
    PROFNOME VARCHAR(50) NOT NULL,
    PROFRUA VARCHAR(100) NOT NULL,
    PROFCIDADE VARCHAR(50) NOT NULL
);
    
CREATE TABLE DISCIPLINA
(
    DISCNUMERO INTEGER PRIMARY KEY NOT NULL AUTO_INCREMENT,
    DISCNOME VARCHAR(50) NOT NULL,
    DISCCURSO VARCHAR(50) NOT NULL,
    QTDAULA INTEGER NOT NULL   
);

CREATE TABLE ALUNO
(
    ALUNUMERO INTEGER PRIMARY KEY NOT NULL AUTO_INCREMENT,
    ALUNOME VARCHAR(50) NOT NULL,
    ALURUA VARCHAR(100) NOT NULL,
    ALUCIDADE VARCHAR(50) NOT NULL
);

CREATE TABLE PROFDISC
(
    ANO INTEGER NOT NULL,
    DISCNUMERO INTEGER NOT NULL,
    PROFNUMERO INTEGER NOT NULL,
    CONSTRAINT DISCNUMEROFK FOREIGN KEY(DISCNUMERO) REFERENCES DISCIPLINA(DISCNUMERO),
    CONSTRAINT PROFNUMEROFK FOREIGN KEY(PROFNUMERO) REFERENCES PROFESSOR(PROFNUMERO)
);

CREATE TABLE MATRICULA
(
    ANO INTEGER NOT NULL,
    DISCNUMERO INTEGER NOT NULL,
    ALUNUMERO INTEGER NOT NULL,
    CONSTRAINT DISCNUMEROFK1 FOREIGN KEY(DISCNUMERO) REFERENCES DISCIPLINA(DISCNUMERO),
    CONSTRAINT ALUNUMEROFK FOREIGN KEY(ALUNUMERO) REFERENCES ALUNO(ALUNUMERO)
);
